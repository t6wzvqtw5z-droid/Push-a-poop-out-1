        function renderProgress(container, count) {
            container.innerHTML = '';
            for (let i=0; i<5; i++) {
                const dot = document.createElement('div');
                dot.className = `dot ${i < count ? 'filled' : ''}`;
                container.appendChild(dot);
            }
        }

        function updateUI() {
            coinEl.textContent = Math.floor(state.coins);
            statClicks.textContent = state.totalClicks;
            let perClick = state.clickMultiplier;
            if (state.ogPurchased) perClick += 5;
            if (state.rainbowPurchased) perClick += 10;
            if (state.boostActive || state.rainbowBoostActive) perClick *= 5;
            statPerClick.textContent = perClick;
            statAuto.textContent = state.autoCoinsPerSec.toFixed(1);
            const elapsed = Math.floor((Date.now()-state.playStartTime)/1000);
            const hours = Math.floor(elapsed/3600), mins = Math.floor((elapsed%3600)/60), secs = elapsed%60;
            statTime.textContent = hours>0 ? `${hours}ч ${mins}м ${secs}с` : `${mins}м ${secs}с`;
            costMul.textContent = state.multiplierCost;
            costAuto.textContent = state.autoclickCost;
            btnMul.disabled = state.coins < state.multiplierCost;
            btnAuto.disabled = state.coins < state.autoclickCost;
            updateBoostUI();
            updateRainbowBoostUI();
            updateCardUI();
            updateSkinGrid();
            clickCore.textContent = state.currentSkin;
            renderProgress(multiplierProgress, state.multiplierBonusCount);
            renderProgress(autoProgress, state.autoBonusCount);
        }

        function updateCardUI() {
            if (state.ogPurchased) {
                ogStatus.textContent = '✅ +5 клик'; buyOgBtn.disabled = true; buyOgBtn.textContent = 'Куплено';
            } else {
                ogStatus.textContent = '💩 10000'; buyOgBtn.disabled = state.coins < 10000; buyOgBtn.textContent = 'Купить';
            }
            if (state.rainbowPurchased) {
                rainbowStatus.textContent = '✅ +10 клик'; buyRainbowBtn.disabled = true; buyRainbowBtn.textContent = 'Куплено';
                rainbowBoostBar.style.display = 'flex';
            } else {
                rainbowStatus.textContent = '💩 15000'; buyRainbowBtn.disabled = state.coins < 15000; buyRainbowBtn.textContent = 'Купить';
                rainbowBoostBar.style.display = 'none';
            }
            if (state.adminCardPurchased) {
                adminCardStatus.textContent = '✅ Пароль: артем58'; buyAdminCardBtn.disabled = true; buyAdminCardBtn.textContent = 'Куплено';
            } else {
                adminCardStatus.textContent = '💩 10000000';
                const allSkinsOwned = state.ownedSkins.length === allSkins.length;
                const allCardsBought = state.ogPurchased && state.rainbowPurchased;
                buyAdminCardBtn.disabled = !(allSkinsOwned && allCardsBought && state.coins >= 10000000);
            }
        }

        function updateSkinGrid() {
            if (!skinGrid) return;
            skinGrid.innerHTML = '';
            const owned = state.ownedSkins || ['🕊️'];
            allSkins.forEach(s => {
                const d = document.createElement('div');
                d.className = `skin-item ${s.emoji === state.currentSkin ? 'equipped' : ''} ${!owned.includes(s.emoji) ? 'locked' : ''}`;
                d.innerHTML = `<span>${s.emoji}</span><div style="font-size:0.7rem;color:#ffcc80;">${s.name}</div>${owned.includes(s.emoji) ? '<div class="skin-owned">✅</div>' : `<div class="skin-price">💩${s.price}</div>`}`;
                d.addEventListener('click', () => {
                    if (!currentUser) { showToast('Сначала зарегистрируйтесь!'); return; }
                    if (owned.includes(s.emoji)) { state.currentSkin = s.emoji; saveUserProgress(); updateUI(); }
                    else if (state.coins >= s.price) { state.coins -= s.price; state.ownedSkins.push(s.emoji); state.currentSkin = s.emoji; saveUserProgress(); updateUI(); }
                    else alert('Недостаточно какашек!');
                });
                skinGrid.appendChild(d);
            });
        }

        function updateBoostUI() {
            const now = Date.now();
            if (state.boostActive && now < state.boostEndTime) {
                const rem = Math.ceil((state.boostEndTime - now)/1000);
                boostTimer.textContent = `${Math.floor(rem/60)}:${(rem%60).toString().padStart(2,'0')}`;
                boostBtn.disabled = true;
            } else {
                state.boostActive = false;
                const cd = 24*60*60*1000;
                const since = now - (state.lastBoostTime||0);
                if (since < cd) {
                    const rem = cd - since;
                    boostTimer.textContent = `${Math.floor(rem/3600000)}ч`;
                    boostBtn.disabled = true;
                } else {
                    boostTimer.textContent = 'Готов';
                    boostBtn.disabled = false;
                }
            }
        }

        function updateRainbowBoostUI() {
            if (!state.rainbowPurchased) return;
            const now = Date.now();
            if (state.rainbowBoostActive && now < state.rainbowBoostEndTime) {
                const rem = Math.ceil((state.rainbowBoostEndTime - now)/1000);
                rainbowBoostTimer.textContent = `${Math.floor(rem/60)}:${(rem%60).toString().padStart(2,'0')}`;
                rainbowBoostBtn.disabled = true;
                document.body.classList.add('rainbow-mode');
                if (!window.rainbowInterval) {
                    window.rainbowInterval = setInterval(spawnRainbowEmoji, 200);
                }
            } else {
                state.rainbowBoostActive = false;
                document.body.classList.remove('rainbow-mode');
                if (window.rainbowInterval) { clearInterval(window.rainbowInterval); window.rainbowInterval = null; }
                const cd = 24*60*60*1000;
                const since = now - (state.lastRainbowBoostTime||0);
                if (since < cd) {
                    const rem = cd - since;
                    rainbowBoostTimer.textContent = `${Math.floor(rem/3600000)}ч`;
                    rainbowBoostBtn.disabled = true;
                } else {
                    rainbowBoostTimer.textContent = 'Готов';
                    rainbowBoostBtn.disabled = false;
                }
            }
        }

        function spawnRainbowEmoji() {
            const emoji = document.createElement('div');
            emoji.className = 'flying-emoji';
            emoji.textContent = '💩';
            emoji.style.left = Math.random()*80 + '%';
            emoji.style.top = Math.random()*80 + '%';
            document.body.appendChild(emoji);
            setTimeout(() => emoji.remove(), 2000);
        }

        function updateLeaderboard() {
            const db = getUsersDB();
            const users = Object.keys(db);
            const stats = users.map(u => {
                const key = `pigeon_progress_${u}`;
                const saved = localStorage.getItem(key);
                let coins = 0;
                if (saved) { try { const d = JSON.parse(saved); coins = d.coins||0; } catch(e) {} }
                return { username: u, coins };
            }).sort((a,b) => b.coins - a.coins);
            leaderList.innerHTML = stats.map((s,i) => `<li class="leader-item"><span>${i+1}. ${s.username}</span><span>💩${Math.floor(s.coins)}</span></li>`).join('');
        }

        function addCoins(amount, el) {
            if (amount <= 0) return;
            state.coins += amount;
            updateUI();
            saveUserProgress();
            if (el) {
                const d = document.createElement('div');
                d.className = 'floating-text';
                d.textContent = `+${amount.toFixed(1)}`;
                const r = el.getBoundingClientRect();
                const cr = clickContainer.getBoundingClientRect();
                d.style.left = (r.left + r.width/2 - cr.left) + 'px';
                d.style.top = (r.top - cr.top) + 'px';
                clickContainer.appendChild(d);
                d.addEventListener('animationend', () => d.remove());
            }
        }

        function getTargetUserKey(username) { return `pigeon_progress_${username}`; }
        giveCoinsBtn.addEventListener('click', () => {
            const target = adminTarget.value.trim();
            const amount = parseInt(adminValue.value);
            if (!target || isNaN(amount)) return showToast('Укажите логин и количество');
            const key = getTargetUserKey(target);
            const saved = localStorage.getItem(key);
            if (!saved) return showToast('Пользователь не найден');
            const data = JSON.parse(saved);
            data.coins = (data.coins||0) + amount;
            localStorage.setItem(key, JSON.stringify(data));
            showToast(`Выдано ${amount} какашек игроку ${target}`);
            if (currentUser?.username === target) { state.coins = data.coins; updateUI(); }
        });

            const amount = parseInt(adminValue.value);
            if (!target || isNaN(amount)) return showToast('Укажите логин и количество');
            const key = getTargetUserKey(target);
            const saved = localStorage.getItem(key);
            if (!saved) return showToast('Пользователь не найден');
            const data = JSON.parse(saved);
            data.clickMultiplier = (data.clickMultiplier||1) + amount;
            localStorage.setItem(key, JSON.stringify(data));

            if (!confirm(`Удалить прогресс игрока ${target}?`)) return;
            localStorage.removeItem(getTargetUserKey(target));
            const db = getUsersDB();
            if (db[target]) { delete db[target]; saveUsersDB(db); }
            showToast(`Аккаунт ${target} крашнут`);
            if (currentUser?.username === target) { logoutUser(); updateUI(); }
        });

        function switchModalTab(tab) {
            skinModal.style.display = tab==='skins' ? 'block' : 'none';
            leaderModal.style.display = tab==='leaders' ? 'block' : 'none';
            accountModal.style.display = tab==='account' ? 'block' : 'none';
            tabPanel.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
            tabPanel.querySelector(`[data-tab="${tab}"]`).classList.add('active');
            if (tab === 'account') {
                accountNameDisplay.textContent = currentUser ? `Вы вошли как ${currentUser.username}` : 'Не вошли';
            }
        }

            if (e.target.classList.contains('tab-btn')) {

            if (!currentUser) {
                // показать окно входа/регистрации? По заданию меню открывается только для авторизованных.
                // Для неавторизованного пользователя сделаем просто вход через prompt? Но оставим старое поведение.
                const u = prompt('Введите логин (если нет - зарегистрируем):');
                if (!u) return;
                const db = getUsersDB();
                if (db[u]) {
                    const p = prompt('Пароль:');
                    if (db[u] === p) { loginUser(u); updateUI(); showToast(`Добро пожаловать, ${u}!`); }
                    else showToast('Неверный пароль');
                } else {

                    db[u] = p;
                    saveUsersDB(db);
                    loginUser(u);
                    state.coins += 100;
                    saveUserProgress();
                    updateUI();
                    showToast(`Аккаунт ${u} создан, +100 какашек!`);
                }

                modalOverlay.classList.add('active');
                updateLeaderboard();
                updateSkinGrid();
                switchModalTab('skins');

        let lastAuto = Date.now();
        function autoTick() {
            const now = Date.now(), delta = (now-lastAuto)/1000;
            if (delta >= 1 && state.autoCoinsPerSec > 0) {
                const earned = state.autoCoinsPerSec * Math.floor(delta);
                if (earned > 0) { state.coins += earned; addCoins(earned, document.querySelector('.score-display')); }
                lastAuto = now - (delta%1)*1000;
                updateUI();
                saveUserProgress();
            }
            requestAnimationFrame(autoTick);
        }

            updateUI();
            if (modalOverlay.classList.contains('active')) updateLeaderboard();
        }, 1000);

        clickCore.addEventListener('click', handleClick);
        clickCore.addEventListener('touchstart', e => { e.preventDefault(); handleClick(); }, {passive:false});

        function init() {
            loadSession();
            if (currentUser) loadUserProgress(); else state = defaultState();
            updateUI();
            lastAuto = Date.now();
            requestAnimationFrame(autoTick);
        }
        init();
